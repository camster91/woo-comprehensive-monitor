```javascript
const EventEmitter = require("events");

const emitter = new EventEmitter();
emitter.setMaxListeners(100);

function emit(event, data) {
  emitter.emit(event, data);
}

function subscribe(callback) {
  const handler = (data) => callback(data);
  emitter.on("sse", handler);
  return () => emitter.off("sse", handler);
}

function broadcast(event, data) {
  emitter.emit("sse", { event, data, timestamp: new Date().toISOString() });
}

module.exports = { emit, subscribe, broadcast };
```

```javascript
const { insert, all, get } = require("../db");

function logAction({ action, user = "anonymous", target = null, details = {} }) {
  return insert(
    `INSERT INTO audit_log (action, user, target, details, timestamp) VALUES (?, ?, ?, ?, datetime('now'))`,
    [action, user, target, JSON.stringify(details)]
  );
}

function getAuditLog({ action, user, limit = 100, offset = 0 } = {}) {
  const conditions = [];
  const params = [];

  if (action) { conditions.push("action = ?"); params.push(action); }
  if (user) { conditions.push("user = ?"); params.push(user); }

  const where = conditions.length ? `WHERE ${conditions.join(" AND ")}` : "";
  const logs = all(
    `SELECT * FROM audit_log ${where} ORDER BY timestamp DESC LIMIT ? OFFSET ?`,
    [...params, limit, offset]
  );
  const countRow = get(`SELECT COUNT(*) as total FROM audit_log ${where}`, params);
  return { logs, total: countRow ? countRow.total : 0 };
}

module.exports = { logAction, getAuditLog };
```

```javascript
const { Router } = require("express");
const path = require("path");
const fs = require("fs");

const router = Router();

router.get("/download/plugin", (req, res) => {
  const pluginDir = path.join(__dirname, "../../plugin-releases");
  if (!fs.existsSync(pluginDir)) {
    return res.status(404).json({ error: "No plugin releases available" });
  }

  const files = fs.readdirSync(pluginDir)
    .filter(f => f.endsWith(".zip"))
    .sort()
    .reverse();

  if (files.length === 0) {
    return res.status(404).json({ error: "No plugin ZIP found" });
  }

  const latest = files[0];
  res.download(path.join(pluginDir, latest), latest);
});

router.get("/download/plugin/info", (req, res) => {
  const pluginDir = path.join(__dirname, "../../plugin-releases");
  if (!fs.existsSync(pluginDir)) {
    return res.json({ available: false });
  }

  const files = fs.readdirSync(pluginDir).filter(f => f.endsWith(".zip")).sort().reverse();
  res.json({
    available: files.length > 0,
    latest: files[0] || null,
    versions: files,
  });
});

module.exports = router;
```

```javascript
const { insert, all, get } = require("../db");

function recordMetrics({ storeId, pageUrl, lcp, fid, cls, ttfb }) {
  return insert(
    `INSERT INTO performance_metrics (store_id, page_url, lcp, fid, cls, ttfb, timestamp)
     VALUES (?, ?, ?, ?, ?, ?, datetime('now'))`,
    [storeId, pageUrl, lcp || null, fid || null, cls || null, ttfb || null]
  );
}

function getMetrics(storeId, { days = 7 } = {}) {
  const cutoff = new Date(Date.now() - days * 24 * 60 * 60 * 1000).toISOString();
  return all(
    `SELECT * FROM performance_metrics WHERE store_id = ? AND timestamp > ? ORDER BY timestamp DESC`,
    [storeId, cutoff]
  );
}

function getDailyAverages(storeId, { days = 7 } = {}) {
  const cutoff = new Date(Date.now() - days * 24 * 60 * 60 * 1000).toISOString();
  return all(
    `SELECT date(timestamp) as date,
            AVG(lcp) as avg_lcp, AVG(fid) as avg_fid,
            AVG(cls) as avg_cls, AVG(ttfb) as avg_ttfb,
            COUNT(*) as samples
     FROM performance_metrics
     WHERE store_id = ? AND timestamp > ?
     GROUP BY date(timestamp)
     ORDER BY date ASC`,
    [storeId, cutoff]
  );
}

module.exports = { recordMetrics, getMetrics, getDailyAverages };
```

```javascript
// Web Vitals collection
(function() {
    if (!window.PerformanceObserver) return;

    var metrics = {};
    var reported = false;

    function sendMetrics() {
        if (reported) return;
        reported = true;
        var xhr = new XMLHttpRequest();
        xhr.open('POST', wcm_tracker.ajax_url || '/wp-json/wcm/v1/report-error');
        xhr.setRequestHeader('Content-Type', 'application/json');
        xhr.send(JSON.stringify({
            type: 'performance_metrics',
            store_id: wcm_tracker.store_id || '',
            page_url: window.location.href,
            lcp: metrics.lcp || null,
            fid: metrics.fid || null,
            cls: metrics.cls || null,
            ttfb: metrics.ttfb || null,
            site: window.location.hostname
        }));
    }

    // TTFB
    try {
        var nav = performance.getEntriesByType('navigation')[0];
        if (nav) metrics.ttfb = nav.responseStart - nav.requestStart;
    } catch(e) {}

    // LCP
    try {
        new PerformanceObserver(function(list) {
            var entries = list.getEntries();
            if (entries.length) metrics.lcp = entries[entries.length - 1].startTime;
        }).observe({ type: 'largest-contentful-paint', buffered: true });
    } catch(e) {}

    // FID
    try {
        new PerformanceObserver(function(list) {
            var entries = list.getEntries();
            if (entries.length) metrics.fid = entries[0].processingStart - entries[0].startTime;
        }).observe({ type: 'first-input', buffered: true });
    } catch(e) {}

    // CLS
    try {
        var clsValue = 0;
        new PerformanceObserver(function(list) {
            list.getEntries().forEach(function(entry) {
                if (!entry.hadRecentInput) clsValue += entry.value;
            });
            metrics.cls = clsValue;
        }).observe({ type: 'layout-shift', buffered: true });
    } catch(e) {}

    // Send after page is loaded and idle
    if (document.readyState === 'complete') {
        setTimeout(sendMetrics, 5000);
    } else {
        window.addEventListener('load', function() { setTimeout(sendMetrics, 5000); });
    }
    // Also send on page hide (covers SPA navigations)
    document.addEventListener('visibilitychange', function() {
        if (document.visibilityState === 'hidden') sendMetrics();
    });
})();
```

**Step 1: Write useEvents hook**

```javascript
import { useEffect, useRef, useCallback, useState } from "react";

export default function useEvents(onEvent) {
  const [connected, setConnected] = useState(false);
  const eventSourceRef = useRef(null);
  const onEventRef = useRef(onEvent);
  onEventRef.current = onEvent;

  useEffect(() => {
    let es;
    let retryTimeout;

    function connect() {
      es = new EventSource("/api/events");
      eventSourceRef.current = es;

      es.onopen = () => setConnected(true);

      es.onmessage = (e) => {
        try {
          const data = JSON.parse(e.data);
          if (onEventRef.current) onEventRef.current(data);
        } catch {}
      };

      es.onerror = () => {
        setConnected(false);
        es.close();
        retryTimeout = setTimeout(connect, 5000);
      };
    }

    connect();

    return () => {
      if (es) es.close();
      if (retryTimeout) clearTimeout(retryTimeout);
    };
  }, []);

  return { connected };
}
```

**Step 2: Write Toast component**

```javascript
import { useState, useCallback, createContext, useContext } from "react";

const ToastContext = createContext();

export function useToast() {
  return useContext(ToastContext);
}

export function ToastProvider({ children }) {
  const [toasts, setToasts] = useState([]);

  const addToast = useCallback((message, type = "info") => {
    const id = Date.now();
    setToasts(prev => [...prev, { id, message, type }]);
    setTimeout(() => {
      setToasts(prev => prev.filter(t => t.id !== id));
    }, 5000);
  }, []);

  return (
    <ToastContext.Provider value={addToast}>
      {children}
      <div className="fixed bottom-4 right-4 z-50 space-y-2">
        {toasts.map(t => (
          <div key={t.id} className={`px-4 py-3 rounded-lg shadow-lg text-sm text-white transition-all ${
            t.type === "error" ? "bg-red-600" :
            t.type === "success" ? "bg-green-600" :
            t.type === "warning" ? "bg-yellow-600" :
            "bg-blue-600"
          }`}>
            {t.message}
          </div>
        ))}
      </div>
    </ToastContext.Provider>
  );
}
```

```jsx
import { useEffect, useState } from "react";
import { api } from "../api/client";

export default function AuditLog() {
  const [data, setData] = useState({ logs: [], total: 0 });
  const [filter, setFilter] = useState({ action: "", user: "" });
  const [loading, setLoading] = useState(true);

  const load = () => {
    const params = new URLSearchParams();
    if (filter.action) params.set("action", filter.action);
    if (filter.user) params.set("user", filter.user);
    api(`/api/audit-log?${params}`).then(d => { setData(d); setLoading(false); });
  };

  useEffect(() => { load(); }, [filter]);

  return (
    <div className="space-y-4">
      <h2 className="text-lg font-semibold">Audit Log ({data.total})</h2>

      <div className="flex gap-2">
        <input placeholder="Filter by action..." value={filter.action}
          onChange={e => setFilter({ ...filter, action: e.target.value })}
          className="px-3 py-1.5 border rounded-lg text-sm" />
        <input placeholder="Filter by user..." value={filter.user}
          onChange={e => setFilter({ ...filter, user: e.target.value })}
          className="px-3 py-1.5 border rounded-lg text-sm" />
      </div>

      {loading ? <p className="text-gray-500">Loading...</p> : (
        <div className="space-y-2">
          {data.logs.length === 0 && <p className="text-gray-400 text-center py-8">No audit entries.</p>}
          {data.logs.map(log => (
            <div key={log.id} className="bg-white rounded-lg p-4 shadow-sm flex items-start gap-3">
              <span className="text-xs px-2 py-0.5 rounded-full bg-gray-100 text-gray-700 whitespace-nowrap font-mono">
                {log.action}
              </span>
              <div className="flex-1 min-w-0">
                <div className="flex gap-3 text-xs text-gray-400">
                  <span>{log.user}</span>
                  {log.target && <span>target: {log.target}</span>}
                  <span>{new Date(log.timestamp).toLocaleString()}</span>
                </div>
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

```jsx
import { Component } from "react";

export default class ErrorBoundary extends Component {
  constructor(props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error) {
    return { hasError: true, error };
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="bg-red-50 border border-red-200 rounded-xl p-6 text-center">
          <h3 className="text-red-700 font-semibold">Something went wrong</h3>
          <p className="text-red-600 text-sm mt-1">{this.state.error?.message}</p>
          <button
            onClick={() => this.setState({ hasError: false, error: null })}
            className="mt-3 px-4 py-2 bg-red-600 text-white rounded-lg text-sm"
          >
            Try Again
          </button>
        </div>
      );
    }
    return this.props.children;
  }
}
```

## Task 13: Database Backup Service

**Files:**
- Create: `server/src/services/backup-service.js`
- Modify: `server/src/index.js`

**Step 1: Write backup-service.js**

```javascript
const fs = require("fs");
const path = require("path");
const { execSync } = require("child_process");
const { saveDB } = require("../db");

const BACKUP_BRANCH = "backups";
const MAX_BACKUPS = 30;

async function runBackup() {
  const dbPath = process.env.DB_PATH || path.join(__dirname, "../data/woo-monitor.db");
  if (!fs.existsSync(dbPath)) {
    console.log("[Backup] No database file found, skipping.");
    return;
  }

  // Force save to disk first
  saveDB();

  const backupDir = path.join(__dirname, "../data/backups");
  if (!fs.existsSync(backupDir)) fs.mkdirSync(backupDir, { recursive: true });

  const timestamp = new Date().toISOString().split("T")[0];
  const backupFile = path.join(backupDir, `woo-monitor-${timestamp}.db`);

  // Copy the database file
  fs.copyFileSync(dbPath, backupFile);
  console.log(`[Backup] Database backed up to ${backupFile}`);

  // Prune old backups
  const backups = fs.readdirSync(backupDir)
    .filter(f => f.endsWith(".db"))
    .sort()
    .reverse();

  if (backups.length > MAX_BACKUPS) {
    backups.slice(MAX_BACKUPS).forEach(f => {
      fs.unlinkSync(path.join(backupDir, f));
      console.log(`[Backup] Pruned old backup: ${f}`);
    });
  }

  // Push to git backups branch if in a git repo
  try {
    const repoRoot = path.join(__dirname, "../..");
    if (fs.existsSync(path.join(repoRoot, ".git"))) {
      execSync(`cd "${repoRoot}" && git stash -q 2>/dev/null; git checkout -B ${BACKUP_BRANCH} 2>/dev/null || true`, { stdio: "pipe" });
      execSync(`cd "${repoRoot}" && cp "${backupFile}" server/data/ && git add server/data/*.db && git commit -m "backup: ${timestamp}" --allow-empty 2>/dev/null || true`, { stdio: "pipe" });
      execSync(`cd "${repoRoot}" && git push origin ${BACKUP_BRANCH} --force 2>/dev/null || true`, { stdio: "pipe" });
      execSync(`cd "${repoRoot}" && git checkout - 2>/dev/null && git stash pop -q 2>/dev/null || true`, { stdio: "pipe" });
      console.log(`[Backup] Pushed to git branch '${BACKUP_BRANCH}'`);
    }
  } catch (err) {
    console.error("[Backup] Git push failed:", err.message);
  }
}

module.exports = { runBackup };
```
